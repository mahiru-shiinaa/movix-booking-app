  // Láº¥y phim theo status
  getMoviesByStatus: async (status) => {
    const response = await fetch(`${BASE_URL}/movies?status=${status}`);

  },

  // TÃ¬m kiáº¿m phim
  searchMovies: async (query) => {
    const response = await fetch(`${BASE_URL}/movies`);
